JS ES6 Modules

The idea is to import and export diferent sections of code from different files into other files which allows you to break appart your code into more smaller grained files which make your code easier to understand

export

There are default exports at the end of the file, or normal export in {}, or export the class inline instead of exporting at the end of the file. And functions can also be exported inline 

export default class User {
  constructor(name, age) {
    this.name = name
    this.age = age
  }
}

export function printName(user) {
  console.log(`User's name is ${user.age}`)
}

export function printAge(user) {
  console.log(`User is ${user.age} years old`)
}

//export default User
//export {printName, printAge}

import

To import a file 

// import relative path
            import User from './user.js'  
// import absolute path
import User from '/user.js'

In order to import we should add type='module' in the script tag in the html file 


  script type="module" src="main.js"

then you can import the user.js file 

export default class User {
  constructor(name, age) {
    this.name = name
    this.age = age
  }
}

export function printName(user) {
  console.log(`User's name is ${user.name}`)
}

export function printAge(user) {
  console.log(`User is ${user.age} years old`)
}

//export default User
//export {printName, printAge}

in the main.js file and use it: 

import User, {printName as printUserName, printAge} from './user.js'

const user = new User('Bob', 11)
console.log(user)
printUserName(user)
printAge(user)

Console

  • User {name: "Bob", age: 11}
  • User´s name is Bob
  • User´s age is 11 years old

To import non default things you write them inside curly braces { printName, printAge }. if you want to change the names you use an alias: { printName as printUserName, printAge }
